from pyhocon import ConfigFactory
from pyhocon.config_tree import ConfigTree
from json import loads
import re

def to_json(config:ConfigTree, level:int=0)->str:
        """Convert HOCON input into a JSON output

        :return: JSON string representation
        :type return: basestring
        """
        lines = ""
        if isinstance(config, ConfigTree):
            if len(config) == 0:
                lines += '{}'
            else:
                lines += '{\n'
                bet_lines = []
                for key, item in config.items():
                    bet_lines.append('{indent}"{key}": {value}'.format(
                        indent=''.rjust((level + 1) * 2, ' '),
                        key=key,
                        value=to_json(item, level + 1))
                    )
                lines += ',\n'.join(bet_lines)
                lines += '\n{indent}}}'.format(indent=''.rjust(level * 2, ' '))
        elif isinstance(config, list):
            if len(config) == 0:
                lines += '[]'
            else:
                lines += '[\n'
                bet_lines = []
                for item in config:
                    bet_lines.append('{indent}{value}'.format(indent=''.rjust((level + 1) * 2, ' '), value=to_json(item, level + 1)))
                lines += ',\n'.join(bet_lines)
                lines += '\n{indent}]'.format(indent=''.rjust(level * 2, ' '))
        elif isinstance(config, str):
            lines = '"{value}"'.format(value=config.replace('\n', '\\n').replace('"', '\\"'))
        elif config is None:
            lines = 'null'
        elif config is True:
            lines = 'true'
        elif config is False:
            lines = 'false'
        else:
            lines = str(config)
        return lines

def parseConfig(location:str,fallback:str=None,convertToJson:bool=False):
    """
    Main function to retrieve the hocon configuration
    :param
    @ location -- config location path
    @ fallback -- fall back location path, default will be None
    @ convertToJson -- converts configTree instance into json file
    :output pyhocon configTree instance or json string repesentation
    """
    if fallback is None:
        config = ConfigFactory.parse_file(location)
    else:
        config =  ConfigFactory.parse_file(location).with_fallback(fallback)
    if convertToJson:
        if isinstance(config,ConfigTree):
            config =  loads(re.sub(r'""', '"', to_json(config)))
    return config
